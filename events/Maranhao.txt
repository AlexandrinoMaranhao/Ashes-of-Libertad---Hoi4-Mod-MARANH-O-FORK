country_event = { #Piaui Frontier Failure
    id = MNH.1
    title = MNH.1.t
    desc = MNH.1.d
    fire_only_once = yes
    is_triggered_only = yes
    picture = x

    option = {
        name = MNH.1.a
		if = {
			limit = {
				CRJ = { has_country_flag = CRJ_emperor_path }
			}
			CRJ = { add_to_faction = MNH }
			else = {
				create_faction = Brazilian_empire_FACTION
			}
		}
		every_country = {
			news_event = {
				id = world.75
				days = 1
			}
		}
		PIA = {
			annex_country = {
				target = VDP
				transfer_troops = yes
			}
			annex_country = {
				target = JIN
				transfer_troops = yes
			}
			annex_country = {
				target = BEP
				transfer_troops = yes
			}
			annex_country = {
				target = PNA
				transfer_troops = yes
			}
		}

		remove_ideas = {
			sarneys_feud
			cangaceiros_light
		}

		add_ideas = {
			sarneys_feud
			mercenary_army_LAM
			cangaceiros_light
		}

		annex_country = {
			target = PIM
			transfer_troops = yes
		}
		
        declare_war_on = {
            target = PIA
            type = annex_everything
        }
        declare_war_on = {
            target = PNA
            type = annex_everything
        }
		
		CFD = {
			country_event = CFD_events.17
		}
		

        set_politics = {
            ruling_party = fascism
        }
		set_popularities = {
			reactionary = 20
			neutrality = 10
			fascism = 70
			communism = 0
			democratic = 0
		}
		
    }
}
