acre_senate_add_seats_1 = {
	custom_effect_tooltip = acrean_senate_1_seat_tt
	hidden_effect = { 
		add_to_variable = { senate_seats = 1 }
		if = {
			limit = { 
				check_variable = { senate_neutral_seats > 0 }
			}
			add_to_variable = { senate_neutral_seats = 1 }
		}
		else_if = {
			limit = {
				AND = {
					check_variable = { senate_militarist_seats > senate_technocrat_seats }
					check_variable = { senate_militarist_seats > senate_nationalist_seats }
				}
			}
			add_to_variable = { senate_militarist_seats = 1 }
		}
		else_if = {
			limit = {
				AND = {
					check_variable = { senate_technocrat_seats > senate_militarist_seats }
					check_variable = { senate_technocrat_seats > senate_nationalist_seats }
				}
			}
			add_to_variable = { senate_technocrat_seats = 1 }
		}
		else_if = {
			limit = {
				AND = {
					check_variable = { senate_nationalist_seats > senate_militarist_seats }
					check_variable = { senate_nationalist_seats > senate_technocrat_seats }
				}
			}
			add_to_variable = { senate_nationalist_seats = 1 }
		}
		else = { add_to_variable = { senate_neutral_seats = 1 } }		
	}
}

acre_senate_add_militarist_seats_1 = {
	custom_effect_tooltip = acrean_senate_militarist_1_seat_tt
	hidden_effect = {
		if = {
			limit = { check_variable = { senate_neutral_seats > 0 } }
			subtract_from_variable = { senate_neutral_seats = 1 }
			add_to_variable = { senate_militarist_seats = 1 }
		}
		else_if = {
			limit = {
				AND = {
					check_variable = { senate_technocrat_seats > 0 }
					check_variable = { senate_nationalist_seats > 0 }
				}
			}
			random_list = {
				50 = {
					subtract_from_variable = { senate_technocrat_seats = 1 }
					add_to_variable = { senate_militarist_seats = 1 }
					modifier = {
						factor = 1.5
						check_variable = { senate_technocrat_seats > senate_nationalist_seats }
					}
				}
				50 = {
					subtract_from_variable = { senate_nationalist_seats = 1 }
					add_to_variable = { senate_militarist_seats = 1 }
					modifier = {
						factor = 1.5
						check_variable = { senate_nationalist_seats > senate_technocrat_seats }
					}
				}
			}
		}
		else_if = {
			limit = {
				AND = {
					check_variable = { senate_technocrat_seats > 0 }
				}	
			}
			subtract_from_variable = { senate_technocrat_seats = 1 }
			add_to_variable = { senate_militarist_seats = 1 }
		}
		else_if = {
			limit = {
				AND = {
					check_variable = { senate_nationalist_seat > 0 }
				}	
			}
			subtract_from_variable = { senate_nationalist_seat = 1 }
			add_to_variable = { senate_militarist_seats = 1 }
		}
	}
}
