para_gui_missions = {
	para_pick_mission = {
		fixed_random_seed = no
		available = {
			hidden_trigger = {
					NOT = { has_country_flag = para_mission0 }
					NOT = { has_country_flag = para_mission1 }
					NOT = { has_country_flag = para_mission2 }
					NOT = { has_country_flag = para_mission3 } }
					custom_trigger_tooltip = { tooltip = para_pick_mission_tt }	 
				}
		visible = { always = yes }
		remove_effect = {
			hidden_effect =  {
				random_list = {
						33 = {
					activate_mission = para_build_military
					set_country_flag = para_mission0
					random_owned_controlled_state = {
						set_variable = { ROOT.mission_target = THIS.id }
						set_variable = { mission_check = arms_factory_level }
						THIS = {save_global_event_target_as  = target_state }
						set_state_flag = {
							flag = new_state
							value = 1
							days = 15
						}	
					}
							}
						33 = {
					activate_mission = para_build_civilian
					set_country_flag = para_mission1
					random_owned_controlled_state = {
						set_variable = { ROOT.mission_target = THIS.id }
						set_variable = { mission_check = industrial_complex_level }
						THIS = {save_global_event_target_as  = target_state }
						set_state_flag = {
							flag = new_state
							value = 1
							days = 15
						}
					}		
							}
						33 = {
					activate_mission = para_build_infrastructure
					set_country_flag = para_mission2
					random_owned_controlled_state = {
						set_variable = { ROOT.mission_target = THIS.id }
						set_variable = { mission_check = infrastructure_level }

						set_state_flag = {
							flag = new_state
							value = 1
							days = 15
						}
					}		
						}	
			}
		}	
	}
		days_remove = 14
	}
