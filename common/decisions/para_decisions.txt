para_gui_missions = {
	para_pick_mission = {
		fixed_random_seed = no
		available = {
			hidden_trigger = {
					NOT = { has_country_flag = para_mission0 }
					NOT = { has_country_flag = para_mission1 }
					NOT = { has_country_flag = para_mission2 }
					NOT = { has_country_flag = para_mission3 } }
					custom_trigger_tooltip = { tooltip = para_pick_mission_tt }	 
				}
		visible = { always = yes }
		remove_effect = {
			hidden_effect =  {
				random_list = {
						33 = {
					activate_mission = para_build_military
					set_country_flag = para_mission0
					random_owned_controlled_state = {
						set_variable = { ROOT.mission_target = THIS.id }
						set_variable = { mission_check = arms_factory_level }
						set_state_flag = {
							flag = new_state
							value = 1
							days = 15
						}	
					}
							}
						33 = {
					activate_mission = para_build_civilian
					set_country_flag = para_mission1
					random_owned_controlled_state = {
						set_variable = { ROOT.mission_target = THIS.id }
						set_variable = { mission_check = industrial_complex_level }
						#save_global_event_target_as = { target_state = THIS }
						set_state_flag = {
							flag = new_state
							value = 1
							days = 15
						}
					}		
							}
						33 = {
					activate_mission = para_build_infrastructure
					set_country_flag = para_mission2
					random_owned_controlled_state = {
						set_variable = { ROOT.mission_target = THIS.id }
						set_variable = { mission_check = infrastructure_level }
						#save_global_event_target_as = { target_state = THIS }
						set_state_flag = {
							flag = new_state
							value = 1
							days = 15
						}
					}		
						}	
			}
		}	
	}
		days_remove = 14
	}
	para_build_military = {
		allowed = { tag = BEM }
		visible = { has_country_flag = para_mission0 tag = BEM }
		available = { var:mission_target = { arms_factory > var:mission_check }}
		activation = { has_country_flag = para_mission0 has_state_flag = new_state }
		selectable_mission = no
		days_mission_timeout = 270
		timeout_effect = {
			subtract_from_variable = { BEM_Prestige = 30 }
			clr_country_flag = para_mission0
			clear_variable = mission_check
		}
		complete_effect = {
			add_to_variable = { BEM_Prestige = 30 }
			clr_country_flag = para_mission0
			clear_variable = mission_check
		}
	}
	para_build_civilian = {
		allowed = { tag = BEM }
		visible = { has_country_flag = para_mission1 tag = BEM }
		available = { var:mission_target = { industrial_complex > var:mission_check } }
		activation = { has_country_flag = para_mission1 has_state_flag = new_state }
		selectable_mission = no
		days_mission_timeout = 270
		timeout_effect = {
			subtract_from_variable = { BEM_Prestige = 30 }
			clr_country_flag = para_mission1
			clear_variable = mission_check1
		}
		complete_effect = {
			add_to_variable = { BEM_Prestige = 30 }
			clr_country_flag = para_mission1
			clear_variable = mission_check1
		}
	}
	para_build_infrastructure = {
		allowed = { tag = BEM }
		visible = { has_country_flag = para_mission2 tag = BEM }
		available = { var:mission_target = { infrastructure > var:mission_check } }
		activation = { has_country_flag = para_mission2 has_state_flag = new_state }
		selectable_mission = no
		days_mission_timeout = 150
		timeout_effect = {
			subtract_from_variable = { BEM_Prestige = 30 }
			clr_country_flag = para_mission2
			clear_variable = mission_check2
		}
		complete_effect = {
			add_to_variable = { BEM_Prestige = 30 }
			clr_country_flag = para_mission2
			clear_variable = mission_check2
		}
	}
}
